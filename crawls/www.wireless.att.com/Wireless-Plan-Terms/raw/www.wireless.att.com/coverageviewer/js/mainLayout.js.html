/** * Selects the National Tab Un-selects the International Tab * */ function showNational() { document.getElementById("nationalTab").className = "current"; document.getElementById("internationalTab").className = ""; document.getElementById("nationalContainer").style.display = "block"; document.getElementById("internationalContainer").style.display = "none"; } /** * Selects the International Tab un-selects the National Tab * */ function showInternational(arURL) { document.getElementById("nationalTab").className = ""; document.getElementById("internationalTab").className = "current"; document.getElementById("nationalContainer").style.display = "none"; document.getElementById("internationalContainer").style.display = "block"; if(document.getElementById('AR_iframe').src == "") document.getElementById('AR_iframe').src = arURL; }
